## üìê Split and Join

### split() - Break String into Array

**Python:**
```python
# Split by whitespace (default)
"Hello World".split()           # ["Hello", "World"]
"  Hello   World  ".split()     # ["Hello", "World"] (handles multiple spaces)

# Split by specific delimiter
"a,b,c".split(",")              # ["a", "b", "c"]
"a,,c".split(",")               # ["a", "", "c"] (empty string preserved)

# Limit number of splits
"a,b,c,d".split(",", 2)         # ["a", "b", "c,d"] (max 2 splits)

# Split every character
list("hello")                    # ["h", "e", "l", "l", "o"]

# Split by newline
"line1\nline2\nline3".split("\n")  # ["line1", "line2", "line3"]
"line1\nline2".splitlines()        # ["line1", "line2"] (handles \r\n too)
```

**JavaScript:**
```javascript
// Split by pattern
"Hello World".split(" ");           // ["Hello", "World"]
"  Hello   World  ".split(" ");     // ["", "", "Hello", "", "", "World", "", ""]

// ‚ö†Ô∏è JS split doesn't trim like Python!
"a,b,c".split(",");                 // ["a", "b", "c"]

// Limit number of splits
"a,b,c,d".split(",", 2);            // ["a", "b"] (keeps first 2 items!)

// Split every character
"hello".split("");                  // ["h", "e", "l", "l", "o"]
[..."hello"];                       // ["h", "e", "l", "l", "o"] (spread)

// Split by regex
"a1b2c3".split(/\d/);               // ["a", "b", "c", ""]
```

### join() - Combine Array into String

**Python:**
```python
# Join with delimiter
",".join(["a", "b", "c"])           # "a,b,c"
" ".join(["Hello", "World"])        # "Hello World"
"".join(["a", "b", "c"])            # "abc"

# Join characters
"".join(['H', 'e', 'l', 'l', 'o'])  # "Hello"

# ‚ö†Ô∏è Can only join strings!
# ",".join([1, 2, 3])               # TypeError!
",".join(map(str, [1, 2, 3]))       # "1,2,3"
",".join(str(x) for x in [1,2,3])   # "1,2,3"
```

**JavaScript:**
```javascript
// Join with delimiter
["a", "b", "c"].join(",");          // "a,b,c"
["Hello", "World"].join(" ");       // "Hello World"
["a", "b", "c"].join("");           // "abc"

// Join automatically converts to strings
[1, 2, 3].join(",");                // "1,2,3"

// Default separator is comma
["a", "b", "c"].join();             // "a,b,c"
```

## üìê Find and Search

### Finding Substrings

**Python:**
```python
s = "Hello World"

# find() - returns index or -1 if not found
s.find("World")         # 6
s.find("xyz")           # -1 (not found)
s.find("o")             # 4 (first occurrence)
s.find("o", 5)          # 7 (search from index 5)
s.rfind("o")            # 7 (last occurrence)

# index() - raises ValueError if not found
s.index("World")        # 6
# s.index("xyz")        # ValueError!

# count() - count occurrences
s.count("o")            # 2
s.count("l")            # 3

# in operator - boolean check
"World" in s            # True
"xyz" in s              # False
```

**JavaScript:**
```javascript
const s = "Hello World";

// indexOf() - returns index or -1 if not found
s.indexOf("World");         // 6
s.indexOf("xyz");           // -1
s.indexOf("o");             // 4
s.indexOf("o", 5);          // 7 (search from index 5)
s.lastIndexOf("o");         // 7

// includes() - boolean check (ES6+)
s.includes("World");        // true
s.includes("xyz");          // false
s.includes("o", 5);         // true (from index 5)

// search() - with regex
s.search(/world/i);         // 6 (case-insensitive)
s.search(/xyz/);            // -1

// match() - returns array of matches
"hello123world".match(/\d+/);      // ["123"]
"a1b2c3".match(/\d/g);             // ["1", "2", "3"]
```

## üìê Replace

### Replacing Substrings

**Python:**
```python
s = "Hello World World"

# replace(old, new) - replaces ALL occurrences
s.replace("World", "Python")        # "Hello Python Python"

# replace(old, new, count) - limit replacements
s.replace("World", "Python", 1)     # "Hello Python World"

# Case-sensitive by default
s.replace("world", "Python")        # "Hello World World" (no change)

# Remove substring (replace with empty)
s.replace("World ", "")             # "Hello World"

# Using regex (re module)
import re
re.sub(r'\d+', 'X', 'a1b22c333')    # "aXbXcX" (replace all numbers)
```

**JavaScript:**
```javascript
const s = "Hello World World";

// replace() - replaces FIRST occurrence only!
s.replace("World", "JS");           // "Hello JS World" ‚ö†Ô∏è

// replaceAll() - replaces ALL occurrences (ES2021+)
s.replaceAll("World", "JS");        // "Hello JS JS"

// Using regex with global flag
s.replace(/World/g, "JS");          // "Hello JS JS"

// Case-insensitive with regex
"HELLO".replace(/hello/i, "hi");    // "hi"

// Remove substring
s.replace(/World /g, "");           // "Hello World"

// Replace with function
"a1b2".replace(/\d/g, n => n * 2);  // "a2b4"
```

## üìê Trim and Strip

### Removing Whitespace

**Python:**
```python
s = "   Hello World   "

# strip() - remove from both ends
s.strip()               # "Hello World"

# lstrip() - remove from left only
s.lstrip()              # "Hello World   "

# rstrip() - remove from right only
s.rstrip()              # "   Hello World"

# Strip specific characters
"###Hello###".strip("#")    # "Hello"
"xxHelloxx".strip("x")      # "Hello"
"abcHelloacb".strip("abc")  # "Hello" (strips any of a, b, c)
```

**JavaScript:**
```javascript
const s = "   Hello World   ";

// trim() - remove from both ends
s.trim();               // "Hello World"

// trimStart() / trimLeft() - remove from left
s.trimStart();          // "Hello World   "

// trimEnd() / trimRight() - remove from right
s.trimEnd();            // "   Hello World"

// No built-in to strip specific chars - use regex
"###Hello###".replace(/^#+|#+$/g, "");  // "Hello"
```

## üìê Prefix and Suffix

### Checking Start/End

**Python:**
```python
s = "Hello World"

# Check prefix
s.startswith("Hello")       # True
s.startswith("hello")       # False (case-sensitive)
s.startswith(("Hi", "Hello"))  # True (tuple of options)

# Check suffix
s.endswith("World")         # True
s.endswith("world")         # False
s.endswith((".py", ".js"))  # False (tuple of options)

# Remove prefix/suffix (Python 3.9+)
"HelloWorld".removeprefix("Hello")  # "World"
"HelloWorld".removesuffix("World")  # "Hello"
```

**JavaScript:**
```javascript
const s = "Hello World";

// Check prefix
s.startsWith("Hello");      // true
s.startsWith("Hello", 0);   // true (from index 0)
s.startsWith("World", 6);   // true (from index 6)

// Check suffix
s.endsWith("World");        // true
s.endsWith("Hello", 5);     // true (checks first 5 chars)

// Remove prefix/suffix (manual)
const removePrefix = (str, prefix) => 
    str.startsWith(prefix) ? str.slice(prefix.length) : str;

const removeSuffix = (str, suffix) => 
    str.endsWith(suffix) ? str.slice(0, -suffix.length) : str;
```

## üìê Character Type Checks

### Checking Character Properties

**Python:**
```python
# Check character type
"Hello".isalpha()       # True (all letters)
"123".isdigit()         # True (all digits)
"Hello123".isalnum()    # True (letters or digits)
"   ".isspace()         # True (all whitespace)
"HELLO".isupper()       # True (all uppercase)
"hello".islower()       # True (all lowercase)

# ‚ö†Ô∏è Empty string returns False for all
"".isalpha()            # False
"".isdigit()            # False

# Check single character
char = 'A'
char.isalpha()          # True
char.isupper()          # True
'5'.isdigit()           # True
```

**JavaScript:**
```javascript
// No built-in methods - use regex or manual checks

// Check if all letters
const isAlpha = str => /^[a-zA-Z]+$/.test(str);
isAlpha("Hello");       // true
isAlpha("Hello123");    // false

// Check if all digits
const isDigit = str => /^\d+$/.test(str);
isDigit("123");         // true

// Check if alphanumeric
const isAlnum = str => /^[a-zA-Z0-9]+$/.test(str);
isAlnum("Hello123");    // true

// Check single character
const isLetter = c => /[a-zA-Z]/.test(c);
const isNumber = c => /\d/.test(c);
const isUpper = c => c === c.toUpperCase() && isLetter(c);
const isLower = c => c === c.toLowerCase() && isLetter(c);
```

## üìê Character Codes

### ASCII/Unicode Conversions

**Python:**
```python
# Character to code
ord('A')            # 65
ord('a')            # 97
ord('0')            # 48

# Code to character
chr(65)             # 'A'
chr(97)             # 'a'

# Common ranges
# 'A'-'Z': 65-90
# 'a'-'z': 97-122
# '0'-'9': 48-57

# Check if lowercase letter
char = 'a'
ord('a') <= ord(char) <= ord('z')  # True

# Convert lowercase to uppercase (manual)
chr(ord('a') - 32)  # 'A'

# Get letter position (0-25)
ord('c') - ord('a')  # 2
ord('C') - ord('A')  # 2
```

**JavaScript:**
```javascript
// Character to code
"A".charCodeAt(0);      // 65
"a".charCodeAt(0);      // 97

// Code to character
String.fromCharCode(65);    // "A"
String.fromCharCode(97);    // "a"

// Get codes for multiple characters
[..."Hello"].map(c => c.charCodeAt(0));  // [72, 101, 108, 108, 111]

// Create string from multiple codes
String.fromCharCode(72, 101, 108, 108, 111);  // "Hello"

// Get letter position (0-25)
"c".charCodeAt(0) - "a".charCodeAt(0);  // 2
```

## üìê Padding and Alignment

**Python:**
```python
s = "Hello"

# Pad to width
s.ljust(10)         # "Hello     " (left-aligned)
s.rjust(10)         # "     Hello" (right-aligned)
s.center(10)        # "  Hello   " (centered)

# Pad with character
s.ljust(10, '-')    # "Hello-----"
s.rjust(10, '0')    # "00000Hello"

# Zero-pad numbers
"42".zfill(5)       # "00042"
"-42".zfill(5)      # "-0042" (sign preserved)
```

**JavaScript:**
```javascript
const s = "Hello";

// Pad to width (ES2017+)
s.padEnd(10);           // "Hello     "
s.padStart(10);         // "     Hello"

// Pad with character
s.padEnd(10, "-");      // "Hello-----"
s.padStart(10, "0");    // "00000Hello"

// Zero-pad numbers
"42".padStart(5, "0");  // "00042"
```

## ‚ö†Ô∏è Common Mistakes

1. **Python replace() vs JavaScript replace()**
   - ‚ùå JS `"aaa".replace("a", "b")` ‚Üí `"baa"` (first only!)
   - ‚úÖ JS `"aaa".replaceAll("a", "b")` ‚Üí `"bbb"`
   - ‚úÖ Python `"aaa".replace("a", "b")` ‚Üí `"bbb"` (all by default)

2. **Split limit behavior differs**
   - Python: `"a,b,c".split(",", 1)` ‚Üí `["a", "b,c"]` (1 split)
   - JS: `"a,b,c".split(",", 1)` ‚Üí `["a"]` (1 result!)

3. **Forgetting method doesn't modify original**
   - ‚ùå `s.upper()` then using `s` (unchanged!)
   - ‚úÖ `s = s.upper()` or `upper_s = s.upper()`

4. **Empty string in contains check**
   - `"" in "hello"` ‚Üí True (Python)
   - `"hello".includes("")` ‚Üí true (JS)
